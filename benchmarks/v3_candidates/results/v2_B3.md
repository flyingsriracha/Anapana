import asyncio, sys
sys.path.insert(0, "/tmp/B3_async_pool_buggy")
from db.pool import ConnectionPool

class S3UploadError(Exception): pass

async def simulate_export_leak(pool):
    conn = await pool.acquire_raw()
    try:
        raise S3UploadError("S3 down")
    except S3UploadError:
        return  # slot leaked

async def fetch_shard(shard_id, pool):
    async with pool.acquire() as conn:
        await asyncio.sleep(0.5)
        return shard_id

async def dashboard_run(pool, label):
    import time; t0 = time.monotonic()
    results = await asyncio.gather(*[fetch_shard(s, pool) for s in range(5)])
    print(f"  [{label}] {len(results)} shards in {time.monotonic()-t0:.2f}s")

async def main():
    healthy = ConnectionPool(max_connections=5)
    # Healthy: all 5 parallel shards in ~0.5s
    await dashboard_run(healthy, "healthy")

    leaky = ConnectionPool(max_connections=5)
    await simulate_export_leak(leaky)
    # 1 slot leaked → 5 shards on 4 slots → 1 shard serialized → ~1.0s per request
    # In prod (1.6s/shard, multiple leaks) → 30s+ total
    await dashboard_run(leaky, "after 1 leak")

asyncio.run(main())

## PROPOSED_FIX
File: `tasks/nightly_export.py`

Change: wrap the `acquire_raw()` call in a `try/finally` so the connection is always released, even when `S3UploadError` causes an early return.

```python
# BEFORE (lines 32-44):
async def export_to_s3():
    conn = await pool.acquire_raw()

    async for chunk in stream_data(conn):
        try:
            await upload_to_s3(chunk)
        except S3UploadError as e:
            logger.error(f"S3 upload failed at chunk {chunk['row']}: {e}")
            return   # <-- LEAKS: release_raw never called on this path

    logger.info("Nightly export complete")
    pool.release_raw(conn)

# AFTER:
async def export_to_s3():
    conn = await pool.acquire_raw()
    try:
        async for chunk in stream_data(conn):
            try:
                await upload_to_s3(chunk)
            except S3UploadError as e:
                logger.error(f"S3 upload failed at chunk {chunk['row']}: {e}")
                return   # now safe: finally block will release
        logger.info("Nightly export complete")
    finally:
        pool.release_raw(conn)
```

Alternative (equivalent correctness): switch to `pool.acquire()` context manager if the cross-coroutine lifetime rationale doesn't apply (it doesn't here — the connection is held within a single coroutine scope).
